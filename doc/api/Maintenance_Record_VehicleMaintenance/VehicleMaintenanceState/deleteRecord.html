void deleteRecord()

Delete the currently selected maintenance record from the list and database.

void deleteRecord() {
  if (selectedRecord != null) {
    setState(() {
      recordDAO.deleteRecord(selectedRecord!);
      maintenanceRecord.remove(selectedRecord);
      selectedRecord = null;
    });

    ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(
          content: Text(AppLocalizations.of(context)!.translate("Record has been deleted permanently")!),
          duration: Duration(seconds: 3),
        )
    );
  }
}
